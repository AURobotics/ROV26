#include <Arduino.h>
#include <TMCStepper.h>
#include <HardwareSerial.h>

#define R_SENSE 0.11f //This is a value that depends on the version of TMC, the TMC2208 silent steptick uses this value
//UART pins for communication:
#define RX_PIN 16 
#define TX_PIN 17
//stepper motor microsteps
#define MS 16 //every step from the motor, contains 16 microsteps
//note: a step is about 1.8 degrees of rotation

TMC2208Stepper driver = TMC2208Stepper(&Serial2, R_SENSE); //create an object from the driver
void setup() {
  Serial2.begin(115200, SERIAL_8N1, RX_PIN, TX_PIN); //begin serial communication with the TMC on the defined serial2 pins
  while(!Serial2){
    delay(1);
  }
  Serial.begin(9600); //begin serial communication with the laptop
  delay(500); 
  driver.begin(); //initialize the TMC driver
  delay(500);
  driver.pdn_disable(true); //allow UART to control (not only configure)
  driver.toff(5); //enable the motor, any value that is not 0 is provided
  driver.pwm_autoscale(true);  //something related to stealthchop, not sure if it is important :)
  driver.I_scale_analog(false); //the TMC has a pin labeled VREF, when used, the current provided to that pin modifies the current being sent to the motor.
  //by setting I_scale_analog to false, the rms current sent to the motor will be controlled using registers, instead of the physical pin
  driver.rms_current(950); //RMS current sent to the motor
  driver.mstep_reg_select(true); //allow modifing the number of microsteps through registers
  driver.microsteps(MS); //set microsteps to MS, which is 16 now
  delay(500);
}


void loop() {
  //VACTUAL takes velocity measure in microsteps/second. since a step is 1.8 degrees, 200 steps is a full 360 degree rotation.
  driver.VACTUAL(MS * 200);  //since we set microsteps to 16, we need 200(steps)*16 to get a full 360 degree rotation per second
  delay(10000); //now the motor is rotating! we will let it rotate for 10 seconds
  driver.VACTUAL(0); //now the motor is stopped
  delay(2000);
}
